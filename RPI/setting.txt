

#RPI 실행시 자동으로 서비스 실행 설정 방법
sudo nano /etc/systemd/system/cam_stream.service (서비스파일이름)
_________________________________________________________________________________

#파일 내부에 작성
[Unit]
Description=Camera stream uploader (설명)
After=network.target

[Service]
Type=simple
User=fbtm
WorkingDirectory=/home/pi/app
ExecStart=/home/fbtm/Descktop/fbm/env/bin/python(가상환경경로)
            /home/fbtm/Desktop/fbtm/cam_stream.py (서비스파일경로)
Restart=always
RestartSec=5
Environment=PYTHONUNBUFFERED=1

[Install]
WantedBy=multi-user.target
________________________________________________________________________________

#활성화 및 실행
sudo systemctl daemon-reload
sudo systemctl enable cam_stream.service(서비스파일이름)
sudo systemctl start cam_stream.service
________________________________________________________________________________

#상태 확인, 로그 보기 
sudo systemctl status cam_stream.service
sudo journalctl -u cam_stream.service
________________________________________________________________________________

#실행 권한 부여
ls -l /home/fbtm/Desktop/fbtm/cam_stream.py (파이썬파일이름)
chmod +x /home/fbtm/Desktop/fbtm/cam_stream.py (파이썬파일이름)
________________________________________________________________________________

#재실행
sudo systemctl restart cam_stream.service (서비스파일이름)
________________________________________________________________________________
